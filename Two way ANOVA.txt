# Two way ANOVA

Two_way_ANOVA <- read_excel("C:/Users/acer/OneDrive/Desktop/AA/Advance_Analytics/Two_way_ANOVA.xlsx")
attach(Two_way_ANOVA)

str(Two_way_ANOVA)

Two_way_ANOVA$Place <- as.factor(Two_way_ANOVA$Place)
Two_way_ANOVA$Education <- as.factor(Two_way_ANOVA$Education)
str(Two_way_ANOVA)

Model <- aov(Sales~Place + Education + Place*Education, data = Two_way_ANOVA)
summary(Model)

TukeyHSD(Model)

model.tables(Model,"mean")

# TO check the residuals
Res <- Model$residuals
Res

shapiro.test(Res)

# TO check homogeneity
library(car)
leveneTest(Sales~Place*Education, data = Two_way_ANOVA)
